

# ⚡ 현장 계산기 통합 데이터 명세서 (전기/전자/통신 총망라)

아래의 JSON 형식 데이터를 바탕으로 계산기의 동적 입력 폼과 결과 출력 로직을 구현해 주세요. 수식은 `formulaLatex`를 사용하여 화면에 렌더링하고, 실제 계산은 `calculate` 함수를 사용합니다.

```javascript
const calculatorRegistry = [
  // ==========================================
  // 1. 기초 전기 (옴의 법칙 및 파생)
  // ==========================================
  {
    id: "calc-voltage", category: "전기 기초", name: "전압 계산",
    inputs: [{ id: "I", label: "전류", unit: "A" }, { id: "R", label: "저항", unit: "Ω" }],
    formulaLatex: "V = I \\times R",
    calculate: (I, R) => I * R
  },
  {
    id: "calc-current", category: "전기 기초", name: "전류 계산",
    inputs: [{ id: "V", label: "전압", unit: "V" }, { id: "R", label: "저항", unit: "Ω" }],
    formulaLatex: "I = \\frac{V}{R}",
    calculate: (V, R) => V / R
  },
  {
    id: "calc-resistance", category: "전기 기초", name: "저항 계산",
    inputs: [{ id: "V", label: "전압", unit: "V" }, { id: "I", label: "전류", unit: "A" }],
    formulaLatex: "R = \\frac{V}{I}",
    calculate: (V, I) => V / I
  },
  {
    id: "calc-joule", category: "전기 기초", name: "주울 효과 (발열량)",
    inputs: [{ id: "I", label: "전류", unit: "A" }, { id: "R", label: "저항", unit: "Ω" }, { id: "t", label: "시간", unit: "초" }],
    formulaLatex: "H = I^2 \\times R \\times t",
    calculate: (I, R, t) => Math.pow(I, 2) * R * t
  },

  // ==========================================
  // 2. 전력 및 교류 회로 (AC Power & Impedance)
  // ==========================================
  {
    id: "calc-active-power", category: "전력 계산", name: "유효 전력 계산",
    inputs: [{ id: "V", label: "전압", unit: "V" }, { id: "I", label: "전류", unit: "A" }, { id: "pf", label: "역률", unit: "cosθ" }],
    formulaLatex: "P = V \\times I \\times \\cos\\theta",
    calculate: (V, I, pf) => V * I * pf
  },
  {
    id: "calc-apparent-power", category: "전력 계산", name: "피상 전력 계산",
    inputs: [{ id: "V", label: "전압", unit: "V" }, { id: "I", label: "전류", unit: "A" }],
    formulaLatex: "S = V \\times I",
    calculate: (V, I) => V * I
  },
  {
    id: "calc-reactive-power", category: "전력 계산", name: "무효 전력 계산",
    inputs: [{ id: "V", label: "전압", unit: "V" }, { id: "I", label: "전류", unit: "A" }, { id: "sin", label: "무효율", unit: "sinθ" }],
    formulaLatex: "Q = V \\times I \\times \\sin\\theta",
    calculate: (V, I, sin) => V * I * sin
  },
  {
    id: "calc-power-factor", category: "전력 계산", name: "역률 계산",
    inputs: [{ id: "P", label: "유효전력", unit: "W" }, { id: "S", label: "피상전력", unit: "VA" }],
    formulaLatex: "\\cos\\theta = \\frac{P}{S}",
    calculate: (P, S) => P / S
  },
  {
    id: "calc-impedance", category: "교류 회로", name: "임피던스 계산 (R-L-C 직렬)",
    inputs: [{ id: "R", label: "저항", unit: "Ω" }, { id: "XL", label: "유도 리액턴스", unit: "Ω" }, { id: "XC", label: "용량 리액턴스", unit: "Ω" }],
    formulaLatex: "Z = \\sqrt{R^2 + (X_L - X_C)^2}",
    calculate: (R, XL, XC) => Math.sqrt(Math.pow(R, 2) + Math.pow(XL - XC, 2))
  },

  // ==========================================
  // 3. 선로 및 케이블 (Wiring & Cables)
  // ==========================================
  {
    id: "calc-volt-drop", category: "선로 계산", name: "전압 강하 계산 (단상)",
    inputs: [{ id: "L", label: "길이", unit: "m" }, { id: "I", label: "전류", unit: "A" }, { id: "A", label: "단면적", unit: "mm²" }],
    formulaLatex: "e = \\frac{35.6 \\times L \\times I}{1000 \\times A}",
    calculate: (L, I, A) => (35.6 * L * I) / (1000 * A)
  },
  {
    id: "calc-max-wire-len", category: "선로 계산", name: "최대 와이어 길이 (ΔV 기준)",
    inputs: [{ id: "e", label: "허용 전압강하", unit: "V" }, { id: "A", label: "단면적", unit: "mm²" }, { id: "I", label: "전류", unit: "A" }],
    formulaLatex: "L = \\frac{1000 \\times e \\times A}{35.6 \\times I}",
    calculate: (e, A, I) => (1000 * e * A) / (35.6 * I)
  },

  // ==========================================
  // 4. 부품 및 소자 계산 (Components)
  // ==========================================
  {
    id: "calc-resistor-series", category: "소자 계산", name: "저항 합계 (직렬)",
    inputs: [{ id: "R1", label: "저항 1", unit: "Ω" }, { id: "R2", label: "저항 2", unit: "Ω" }],
    formulaLatex: "R_t = R_1 + R_2",
    calculate: (R1, R2) => Number(R1) + Number(R2)
  },
  {
    id: "calc-resistor-parallel", category: "소자 계산", name: "저항 합계 (병렬)",
    inputs: [{ id: "R1", label: "저항 1", unit: "Ω" }, { id: "R2", label: "저항 2", unit: "Ω" }],
    formulaLatex: "R_t = \\frac{R_1 \\times R_2}{R_1 + R_2}",
    calculate: (R1, R2) => (R1 * R2) / (Number(R1) + Number(R2))
  },
  {
    id: "calc-capacitor-parallel", category: "소자 계산", name: "콘덴서 합계 (병렬)",
    inputs: [{ id: "C1", label: "콘덴서 1", unit: "F" }, { id: "C2", label: "콘덴서 2", unit: "F" }],
    formulaLatex: "C_t = C_1 + C_2",
    calculate: (C1, C2) => Number(C1) + Number(C2)
  },
  {
    id: "calc-led-resistor", category: "소자 계산", name: "LED 보호 저항 계산",
    inputs: [{ id: "Vs", label: "공급 전압", unit: "V" }, { id: "Vf", label: "LED 전압", unit: "V" }, { id: "If", label: "LED 전류", unit: "A" }],
    formulaLatex: "R = \\frac{V_s - V_f}{I_f}",
    calculate: (Vs, Vf, If) => (Vs - Vf) / If
  },
  {
    id: "calc-transformer", category: "기기 계산", name: "변압기 권선비 계산 (2차 전압)",
    inputs: [{ id: "V1", label: "1차 전압", unit: "V" }, { id: "N1", label: "1차 권선수", unit: "회" }, { id: "N2", label: "2차 권선수", unit: "회" }],
    formulaLatex: "V_2 = V_1 \\times \\frac{N_2}{N_1}",
    calculate: (V1, N1, N2) => V1 * (N2 / N1)
  },

  // ==========================================
  // 5. 기타 실무 및 시스템 (Misc & Systems)
  // ==========================================
  {
    id: "calc-battery-life", category: "기타 계산", name: "배터리 수명 계산",
    inputs: [{ id: "C", label: "배터리 용량", unit: "mAh" }, { id: "I", label: "소비 전류", unit: "mA" }],
    formulaLatex: "T = \\frac{C}{I} \\times 0.7 \\text{ (안전율 70\\%)}",
    calculate: (C, I) => (C / I) * 0.7
  },
  {
    id: "calc-antenna-len", category: "통신 계산", name: "안테나 길이 (1/4 파장)",
    inputs: [{ id: "f", label: "주파수", unit: "MHz" }],
    formulaLatex: "L = \\frac{300}{f} \\times \\frac{1}{4}",
    calculate: (f) => (300 / f) * 0.25
  },
  {
    id: "calc-cctv-storage", category: "시스템 계산", name: "CCTV 하드드라이브 용량",
    inputs: [{ id: "bitrate", label: "비트레이트", unit: "Mbps" }, { id: "hours", label: "녹화 시간", unit: "시간" }],
    formulaLatex: "\\text{Storage (GB)} = \\frac{\\text{Bitrate} \\times 3600 \\times \\text{Hours}}{8 \\times 1024}",
    calculate: (bitrate, hours) => (bitrate * 3600 * hours) / (8 * 1024)
  }
];
```

* 이 코드를 그대로 사용하지 않아도 상관없음. 그대신 여기에 있는 공식들을 직접 판단하여 수정할건 수정해서 최종적으로 수식을 대입해서 계산할 수 있도록 할 것.
* 코드 내에 카테고리라고 되어 있는 것은 빼고 name: 뒤에 있는 이름들을 본 떠 내 전기공사 계산기 내부 카테고리에 이름을 하나하나 등록해 수식을 입력하고 계산 할 수 있도록 할 것.
